PROGRAM:

#include <stdio.h>
int main() {
    int n, i, j;
    printf("Enter number of processes: ");
    scanf("%d", &n);

    int bt[n], wt[n], tat[n], p[n], temp;

    printf("Enter Burst Time:\n");
    for(i = 0; i < n; i++) {
    	printf("Process %d:",i);
        scanf("%d",&bt[i]);
        p[i] = i + 1;
    }

    for(i = 0; i < n-1; i++) {
        for(j = i+1; j < n; j++) {
            if(bt[i] > bt[j]) {
                temp = bt[i]; bt[i] = bt[j]; bt[j] = temp;
                temp = p[i];  p[i] = p[j];   p[j] = temp;
            }
        }
    }

    wt[0] = 0;
    for(i = 1; i < n; i++)
        wt[i] = wt[i-1] + bt[i-1];

    for(i = 0; i < n; i++)
        tat[i] = wt[i] + bt[i];

    printf("\nProcess   BT     WT    TAT\n");
    for(i = 0; i < n; i++)
        printf("P %d\t %d\t %d\t %d\n",   p[i], bt[i], wt[i], tat[i]);

    return 0;
}
